Sum Of Diagonal Of Matrix

 

import java.util.Scanner;
class SumofDiagoOfMatrix{
          public static void main(String args[]){
                    Scanner sc = new Scanner(System.in);
                    System.out.println("Enter the size of square matrix :");
                    int size = sc.nextInt();
                    int lsum=0,rsum=0;
                    int a[][] = new int [size][size];
//get input of first matrix from user
                    System.out.println("Enter the matrix elements :");
                    for(int i=0;i<size;i++){
                         for(int j=0;j<size;j++){
                               a[i][j]=sc.nextInt();
                         }
                    }
//calculate cum of left and right diagonal
                    for(int i=0;i<size;i++){
                          for(int j=0;j<size;j++){
                                   if(i==j)                      //condition for left diagonal
                                             lsum=lsum+a[i][j];
                                   else if((i+j)==(size-1)) // condition for right diagonal
                                              rsum=rsum+a[i][j];
                          }
                     }
                     System.out.println("Left Diagonal sum = "+lsum); 
                     System.out.println("Right Diagonal sum = "+rsum);
          }
}
